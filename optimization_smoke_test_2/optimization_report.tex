\section*{General Equations}
\subsection*{Geometry}
\[
b(x)=\begin{cases}0.7545 + (0.1126)x, & 0 \le x \le 4.0000\\1.2051 + (-0.0107)(x-4.0000), & 4.0000 < x \le 8.0000\end{cases}
\]
\[
h(x)=\begin{cases}0.7000 + (0.0015)x, & 0 \le x \le 4.0000\\0.7060 + (0.1290)(x-4.0000), & 4.0000 < x \le 8.0000\end{cases}
\]
\[
A(x)=b(x)h(x)-[b(x)-2t][h(x)-2t]
\]
\[
I_{upright}(x)=\frac{b(x)h(x)^3-[b(x)-2t][h(x)-2t]^3}{12}
\]
\[
I_{side}(x)=\frac{h(x)b(x)^3-[h(x)-2t][b(x)-2t]^3}{12}
\]
\[
S(x)=\frac{I(x)}{c(x)}
\]

\subsection*{Statics, Stress, and Deflection}
\[
R_A=\frac{P(L-a)}{L}, \qquad R_B=\frac{Pa}{L}
\]
\[
\sigma_b(x)=\frac{|M(x)|}{S(x)}
\]
\[
\tau(x)=\frac{|V(x)|}{A_{shear}(x)}
\]
\[
\sigma_{vm}(x)=\sqrt{\sigma_b(x)^2+3\tau(x)^2}
\]
\[
FoS(x)=\frac{\sigma_{allow,eff}}{\sigma_{vm}(x)}
\]
\[
E_{eff}I(x)v''(x)=M(x)
\]
\[
W=\rho\int_0^L A(x)\,dx
\]
