\section*{General Equations}
\subsection*{Geometry}
\[
b(x)=\begin{cases}0.3500 + (0.0000)x, & 0 \le x \le 4.0000\\0.3500 + (0.0405)(x-4.0000), & 4.0000 < x \le 8.0000\end{cases}
\]
\[
h(x)=\begin{cases}0.8089 + (-0.1072)x, & 0 \le x \le 4.0000\\0.3800 + (0.0902)(x-4.0000), & 4.0000 < x \le 8.0000\end{cases}
\]
\[
r(x)=\begin{cases}0.7287 + (-0.0768)x, & 0 \le x \le 4.0000\\0.4213 + (0.0085)(x-4.0000), & 4.0000 < x \le 8.0000\end{cases}
\]
\[
A(x)=2b(x)t+[1-r(x)]2t[h(x)-2t]
\]
\[
I_{upright}(x)=\frac{b(x)h(x)^3-[b(x)-2t][h(x)-2t]^3}{12}-2\frac{t[r(x)(h(x)-2t)]^3}{12}
\]
\[
I_{side}(x)=\frac{h(x)b(x)^3-[h(x)-2t][b(x)-2t]^3}{12}-2\frac{t[r(x)(b(x)-2t)]^3}{12}
\]
\[
S(x)=\frac{I(x)}{c(x)}
\]

\subsection*{Statics, Stress, and Deflection}
\[
R_A=\frac{P(L-a)}{L}, \qquad R_B=\frac{Pa}{L}
\]
\[
\sigma_b(x)=\frac{|M(x)|}{S(x)}
\]
\[
\tau(x)=\frac{|V(x)|}{A_{shear}(x)}, \qquad A_{shear}(x)=[1-r(x)]2t\,h_{clear}(x)
\]
\[
\sigma_{vm}(x)=\sqrt{\sigma_b(x)^2+3\tau(x)^2}
\]
\[
FoS(x)=\frac{\sigma_{allow,eff}}{\sigma_{vm}(x)}
\]
\[
E_{eff}I(x)v''(x)=M(x)
\]
\[
W=\rho\int_0^L A(x)\,dx
\]

\section*{Automatic Equation Substitution}
\subsection*{Effective Material Properties}
\[E_{eff} = E \cdot k_{orientation} \cdot k_{infill} = 500000.0 \cdot 0.700 \cdot 0.850 = 297500.0\ \mathrm{psi}\]
\[\sigma_{allow,eff} = \sigma_{allow} \cdot k_{orientation} \cdot k_{infill} = 4000.0 \cdot 0.700 \cdot 0.850 = 2380.0\ \mathrm{psi}\]
\[W = \rho \int_0^L A(x)\,dx = 0.02631\ \mathrm{lbf}\]
Global governing factor of safety is the minimum value over all beam stations and over both load cases.
\subsection*{Load Case 1}
Active bending orientation: upright (upright section properties).
\[R_A = \frac{P(L-a)}{L} = \frac{5.000(8.000-4.000)}{8.000} = 2.500\ \mathrm{lbf}\]
\[R_B = \frac{Pa}{L} = \frac{5.000(4.000)}{8.000} = 2.500\ \mathrm{lbf}\]
\[FoS_{min} = \frac{\sigma_{allow,eff}}{\sigma_{vm,max}} = \frac{2380.0}{1538.3} = 1.547\]
\[x_{FoS,min} = 4.000\ \mathrm{in}\]
\[\max |v(x)| = 0.08488\ \mathrm{in}, \qquad \max |\theta(x)| = 0.027769\ \mathrm{rad}\]
\subsection*{Load Case 2}
Active bending orientation: side (side-rotated section properties).
